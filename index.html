/* ================= THREE ================= */ const scene = new THREE.Scene(); scene.background = new THREE.Color(0x87ceeb); const camera = new THREE.PerspectiveCamera(75, innerWidth/innerHeight, 0.1, 1000); const renderer = new THREE.WebGLRenderer(); renderer.setSize(innerWidth, innerHeight); document.body.appendChild(renderer.domElement); /* ground */ const ground = new THREE.Mesh( new THREE.PlaneGeometry(200,200), new THREE.MeshStandardMaterial({ color: 0x3cb043 }) ); ground.rotation.x = -Math.PI/2; scene.add(ground); scene.add(new THREE.AmbientLight(0xffffff, 1)); /* ================= PLAYER ================= */ function createPlayer(color){ const g = new THREE.Group(); const body = new THREE.Mesh( new THREE.BoxGeometry(1,1.5,0.5), new THREE.MeshStandardMaterial({ color }) ); body.position.y = 1; g.add(body); const head = new THREE.Mesh( new THREE.SphereGeometry(0.4), new THREE.MeshStandardMaterial({ color: 0xffccaa }) ); head.position.y = 2.1; g.add(head); return g; } const me = createPlayer(0x0066ff); scene.add(me); camera.position.y = 1.7; /* ================= MOBILE JOYSTICK ================= */ let joy = { x:0, y:0 }; const stick = document.getElementById("stick"); const joyBox = document.getElementById("joy"); let dragging = false; joyBox.addEventListener("touchstart", e => dragging = true); joyBox.addEventListener("touchend", e => { dragging = false; joy = { x:0, y:0 }; stick.style.transform = `translate(0px,0px)`; }); joyBox.addEventListener("touchmove", e => { if(!dragging) return; let t = e.touches[0]; let rect = joyBox.getBoundingClientRect(); let dx = t.clientX - (rect.left + rect.width/2); let dy = t.clientY - (rect.top + rect.height/2); let max = 40; dx = Math.max(-max, Math.min(max, dx)); dy = Math.max(-max, Math.min(max, dy)); joy.x = dx / max; joy.y = dy / max; stick.style.transform = `translate(${dx}px,${dy}px)`; }); /* ================= LOOK (drag screen) ================= */ let yaw = 0; document.addEventListener("touchmove", (e)=>{ if(e.target.id === "joy" || e.target.id === "stick") return; yaw -= e.touches[0].movementX ? e.touches[0].movementX * 0.002 : 0.01; }); /* fallback mouse */ document.addEventListener("mousemove", (e)=>{ yaw -= e.movementX * 0.002; }); /* ================= MOVEMENT ================= */ function updatePlayer(){ let speed = 0.1; let forwardX = Math.sin(yaw); let forwardZ = Math.cos(yaw); me.position.x += forwardX * -joy.y * speed; me.position.z += forwardZ * -joy.y * speed; me.position.x += Math.sin(yaw + Math.PI/2) * joy.x * speed; me.position.z += Math.cos(yaw + Math.PI/2) * joy.x * speed; me.rotation.y = yaw; camera.position.set( me.position.x, me.position.y + 0.7, me.position.z ); camera.rotation.y = yaw; socket.emit("move", me.position); } /* ================= SHOOT ================= */ document.getElementById("fire").addEventListener("touchstart", ()=>{ socket.emit("shoot", me.position); }); /* desktop fallback */ document.getElementById("fire").addEventListener("click", ()=>{ socket.emit("shoot", me.position); }); /* ================= MULTIPLAYER ================= */ let players = {}; function createRemote(){ const p = createPlayer(0xff4444); scene.add(p); return p; } socket.on("init", data=>{ for(let id in data){ if(id !== socket.id){ players[id] = createRemote(); } } }); socket.on("newPlayer", d=>{ players[d.id] = createRemote(); }); socket.on("move", d=>{ if(players[d.id]){ players[d.id].position.set(d.data.x,0,d.data.z); } }); /* ================= LOOP ================= */ function animate(){ requestAnimationFrame(animate); updatePlayer(); renderer.render(scene,camera); } animate();
